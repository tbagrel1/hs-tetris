    8 module Coords
    9   ( RatCoord,
   10     IntCoord,
   11     addCoord,
   12     toIntCoord,
   13   )
   14 where
   15 
   16 -- | Type alias for rational coordinates
   17 type RatCoord = (Rational, Rational)
   18 
   19 -- | Type alias for integer coordinates
   20 type IntCoord = (Int, Int)
   21 
   22 -- | Translates the first point using the second one as a movement vector.
   23 addCoord ::
   24   Num a =>
   25   -- | First point
   26   (a, a) ->
   27   -- | Second point
   28   (a, a) ->
   29   -- | Result of the translation
   30   (a, a)
   31 addCoord (x1, y1) (x2, y2) = (x1 + x2, y1 + y2)
   32 
   33 -- | Rounds the point coordinates to the nearest integers to give a IntCoord
   34 toIntCoord ::
   35   -- | Rational coordinates to round
   36   RatCoord ->
   37   -- | Result of the rounding
   38   IntCoord
   39 toIntCoord (x, y) = (round x, round y)
